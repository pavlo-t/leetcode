Redundant Connection
===

```java
class Solution {
    class UnionFind {
        int[] parent;
        int[] rank;
        int size;

        public UnionFind(int size) {
            parent = new int[size];
            for (int i = 0; i < size; i++) {
                parent[i] = i;
            }
            rank = new int[size];
            this.size = size;
        }

        public int find(int i) {
            if (parent[i] != i) {
                 parent[i] = find(parent[i]);
            }
            return parent[i];
        }

        public boolean union(int x, int y) {
            int xp = find(x);
            int yp = find(y);
            if (xp == yp) {
                return false;
            }
            if (rank[xp] < rank[yp]) {
                parent[xp] = yp;
            } else if (rank[xp] > rank[yp]) {
                parent[yp] = xp;
            } else {
                parent[yp] = xp;
                rank[xp]++;
            }

            return true;
        }
    }
    public int[] findRedundantConnection(int[][] edges) {
        //  int N = 1001;
        int N = edges.length;
        UnionFind uf = new UnionFind(N);
        for (int[] edge: edges) {
            if (!uf.union(edge[0] - 1, edge[1] - 1)) {
                return edge;
            }
        }
        return new int[0];
    }

}
```

```scala
object Solution {
  def findRedundantConnection(edges: Array[Array[Int]]): Array[Int] = {
    val uf = new UF(edges.length)
    for (e <- edges) {
      if (!uf.union(e(0) - 1, e(1) - 1))
      return e
    }
    Array()
  }

  private class UF(size: Int) {
    val parent = (0 until size).toArray
    val rank = Array.ofDim(size)

    def find(x: Int): Int = {
      if (parent(x) != x)
        parent(x) = find(parent(x))
      find(parent(x))
    }
    def union(a: Int, b: Int): Boolean = {
      val ap = find(a)
      val bp = find(b)
      if (ap != bp) false else {
        if (rank(ap) < rank(bp)) {
          parent(ap) = bp
        } else if (rank(ap) > rank(bp)) {
          parent(bp) = ap
        } else {
          parent(bp) = ap
          rank(ap) += 1
        }
        true
      }
    }
  }
}
```
