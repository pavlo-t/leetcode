24 Out of Boundary Path
===

```java
class Solution {
  public int findPaths(
    int m, int n, int maxMove,
    int startRow, int startColumn
  ) {
    if (maxMove == 0) return 0;
    int[][] dpCurr = new int[m+2][n+2], dpLast = new int[m+2][n+2];
    for (int i = 1; i <= m; i++) {
      dpCurr[i][1]++;
      dpCurr[i][n]++;
    }
    for (int j = 1; j <= n; j++) {
      dpCurr[1][j]++;
      dpCurr[m][j]++;
    }
    int ans = dpCurr[startRow+1][startColumn+1];
    for (int d = 1; d < maxMove; d++) {
      int[][] temp = dpCurr;
      dpCurr = dpLast;
      dpLast = temp;
      for (int i = 1; i <= m; i++)
        for (int j = 1; j <= n; j++)
          dpCurr[i][j] = (int)(((long)dpLast[i-1][j] + dpLast[i+1][j] + dpLast[i][j-1] + dpLast[i][j+1]) % 1000000007L);
      ans = (ans + dpCurr[startRow+1][startColumn+1]) % 1000000007;
    }
    return ans;
  }
}
```

```scala
object Solution {
  def findPaths(
    m: Int, n: Int,
    maxMove: Int,
    sr: Int, sc: Int
  ): Int = if (maxMove == 0) 0 else {
    val dp0 =
      Array.ofDim[Int](m + 2, n + 2)
    val dp1 =
      Array.ofDim[Int](m + 2, n + 2)
    for (i <- 1 to m) {
      dp1(i)(1) += 1
      dp1(i)(n) += 1
    }
    for (i <- 1 to n) {
      dp1(1)(i) += 1
      dp1(m)(i) += 1
    }
    var r = dp1(sr + 1)(sc + 1)

    for (_ <- 1 until maxMove) {
      val tmp = dp1
      dp1 = dp0
      dp0 = tmp
      for (i <- 1 to m; j <- 1 to n)
        dp1(i)(j) = ((
          dp0(i - 1)(j).toLong
          + dp0(i + 1)(j).toLong
          + dp0(i)(j - 1).toLong
          + dp0(i)(j + 1).toLong
        ) % 1_000_000_007L).toInt
      r = (r + dp1(sr + 1)(sc + 1)) % 1_000_000_007
    }

    r
  }
}
```
